311  void setupActions() {

315  if (sd_opt) {

320  cfg.sd_active = false;
321  }
322  // pin conflicts for pinInputMode() with the SD CS pin for AIThinker and
323  // buttons
324  int input_mode = pinInputMode();
325  if (input_mode != -1 && (input_mode != sd_cs || !cfg.sd_active))
326  addAction(pinInputMode(), actionStartStop);
327 
328  // pin conflicts with AIThinker A101: key6 and headphone detection
329  int head_phone = pinHeadphoneDetect();
330  if (head_phone != -1 && (getPinID(PinFunction::KEY, 6) != head_phone)) {
331  actions.add(pinHeadphoneDetect(), actionHeadphoneDetection,
332  AudioActions::ActiveChange);
333  }
334 
335  // pin conflicts with SD Lyrat SD CS GpioPin and buttons / Conflict on Audiokit
336  // V. 2957
337  int vol_up = pinVolumeUp();
338  int vol_down = pinVolumeDown();
339  if ((vol_up != -1 && vol_down!=-1) && (!cfg.sd_active ||
340  (vol_down != sd_cs && vol_up != sd_cs))) {
341  LOGD("actionVolumeDown")
342  addAction(vol_down, actionVolumeDown);
343  LOGD("actionVolumeUp")
344  addAction(vol_up, actionVolumeUp);
345  } else {
346  LOGW("Volume Buttons ignored because of conflict: %d ", pinVolumeDown());
