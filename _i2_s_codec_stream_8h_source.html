163  codec_cfg.i2s.bits = toCodecBits(info.bits_per_sample);
164  codec_cfg.i2s.rate = toRate(info.sample_rate);
165  codec_cfg.i2s.fmt = toFormat(info.i2s_format);
166  // use reverse logic for codec setting

168  if (p_board == nullptr)
