520  void flushMixer() {
521  LOGD("flush");
522  bool result = false;
523 
524  // determine ringbuffer with mininum available data
525  size_t samples = size_bytes/sizeof(T);
526  for (int j=0;j<output_count;j++){
527  samples = MIN(samples, (size_t) buffers[j]->available());
528  }
529 
530  if (samples>0){
531  result = true;
532  // mix data from ringbuffers to output
533  output.resize(samples);
534  memset(output.data(),0, samples*sizeof(T));
535  for (int j=0;j<output_count;j++) {
536  float weight = weights[j];
537  // sum up input samples to result samples
538  for (int i=0;i<samples; i++){
539  output[i] += weight * buffers[j]->read() / total_weights;
540  }
541  }
542

544  LOGD("write to final out: %d", samples*sizeof(T));
545  p_final_output->write((uint8_t*)output.data(), samples*sizeof(T));
546  }
547  stream_idx = 0;
548  return;
549  }
