39 public:

41  p_buffer = &buffer;
42  p_mutex = &mutex;
43  }
44 
45  // reads a single value
46  T read() override {
47  TRACED();
48  LockGuard guard(p_mutex);
49  return p_buffer->read();
50  }
51 
52  // reads multiple values
53  int readArray(T data[], int len) {
54  TRACED();
55  LockGuard guard(p_mutex);
56  int lenResult = MIN(len, available());
57  for (int j = 0; j < lenResult; j++) {
58  data[j] = p_buffer->read();
59  }
60  return lenResult;
61  }

63  int writeArray(const T data[], int len) {
64  LOGD("%s: %d", LOG_METHOD, len);
65  LockGuard guard(p_mutex);
66  int result = 0;
67  for (int j = 0; j < len; j++) {
68  if (p_buffer->write(data[j]) == 0) {
69  break;
70  }
71  result = j + 1;
72  }
73  return result;
74  }
