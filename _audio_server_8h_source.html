213  void processClient() {
214  //LOGD("processClient");
215  if (client_obj) { // if you get a client,
216  LOGI("New Client."); // print a message out the serial port
217  String currentLine = ""; // make a String to hold incoming data from the client
218  while (client_obj.connected()) { // loop while the client's connected
219  if (client_obj.available()) { // if there's bytes to read from the client,
220  char c = client_obj.read(); // read a byte, then
221  if (c == '\n') { // if the byte is a newline character
222 
223  // if the current line is blank, you got two newline characters in a row.
224  // that's the end of the client HTTP request, so send a response:
225  if (currentLine.length() == 0){
226  sendReplyHeader();
227  sendReplyContent();
228  // break out of the while loop:
229  break;
230  } else { // if you got a newline, then clear currentLine:
231  currentLine = "";
232  }
233  }
234  else if (c != '\r') { // if you got anything else but a carriage return character,
235  currentLine += c; // add it to the end of the currentLine
236  }
237  }
238  }
239  }
