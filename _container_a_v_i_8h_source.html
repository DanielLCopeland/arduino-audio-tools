295 class ParseBuffer {
296  public:
297  size_t writeArray(uint8_t *data, size_t len) {
298  int to_write = min(availableToWrite(), (int)len);
299  memmove(vector.data() + available_byte_count, data, to_write);
300  available_byte_count += to_write;
301  return to_write;
302  }
303  void consume(int size) {
304  memmove(vector.data(), &vector[size], available_byte_count - size);
305  available_byte_count -= size;
306  // memset(&vector[size], 0, size);
307  }
308  void resize(int size) { vector.resize(size + 4); }
309 
310  uint8_t *data() { return vector.data(); }
311 
312  int availableToWrite() { return size() - available_byte_count; }
313 
314  int available() { return available_byte_count; }
315 
316  void clear() {
317  available_byte_count = 0;
318  memset(vector.data(), 0, vector.size());
319  }
320 
321  bool isEmpty() { return available_byte_count == 0; }
322 
323  int size() { return vector.size(); }
324 
325  int indexOf(const char *str) {
326  uint8_t *ptr = (uint8_t *)memmem(vector.data(), available_byte_count, str,
327  strlen(str));
328  return ptr == nullptr ? -1 : ptr - vector.data();
329  }
330 
331  protected:
332  Vector<uint8_t> vector{0};
333  int available_byte_count = 0;
334 };
