312 class AudioVideoSync {
313  public:
314  void setOutput(AudioOutput *out) { p_out = out; }
315  void writeAudio(uint8_t *data, size_t size) {
316  // make space in buffer by playing the audio that is not fitting
317  int to_write = buffer.availableForWrite();
318  if (to_write < size) {
319  int to_read = size - to_write;
320  uint8_t bytes[to_read];
321  buffer.readArray(bytes, to_read);
322  p_out->write(bytes, to_read);
323  }
324  // store audio to buffer
325  buffer.writeArray(data, size);
326  }
327  void resize(int size) { buffer.resize(size); }
328 
329  void setMicrosecondsPerFrame(int32_t time) { us_per_frame = time; }
330 
331  void delay(uint32_t time_used_ms) {
332  int32_t delay_ms = us_per_frame / 1000 - time_used_ms;
333  uint64_t timeout = millis() + delay_ms;
334  while (millis() < timeout) {
335  uint8_t bytes[80];
336  int byte_count = buffer.readArray(bytes, 80);
337  int written = p_out->write(bytes, byte_count);
338  // if nothing to write we add a delay
339  if (written == 0) {
340  delay(10);
341  }
342  }
343  }
344 
345  protected:
346  RingBuffer<uint8_t> buffer{1024 * 20};
347  int32_t us_per_frame;
348  AudioOutput *p_out = nullptr;
349 };

351 class ParseBuffer {
352  public:
353  size_t writeArray(uint8_t *data, size_t len) {
354  int to_write = min(availableToWrite(), (size_t)len);
355  memmove(vector.data() + available_byte_count, data, to_write);
356  available_byte_count += to_write;
357  return to_write;
358  }
359  void consume(int size) {
360  memmove(vector.data(), &vector[size], available_byte_count - size);
361  available_byte_count -= size;
362  // memset(&vector[size], 0, size);
363  }
364  void resize(int size) { vector.resize(size + 4); }
365 
366  uint8_t *data() { return vector.data(); }
367 
368  size_t availableToWrite() { return size() - available_byte_count; }
369 
370  size_t available() { return available_byte_count; }
371 
372  void clear() {
373  available_byte_count = 0;
374  memset(vector.data(), 0, vector.size());
375  }
376 
377  bool isEmpty() { return available_byte_count == 0; }
378 
379  size_t size() { return vector.size(); }
380 
381  long indexOf(const char *str) {
382  uint8_t *ptr = (uint8_t *)memmem(vector.data(), available_byte_count, str,
383  strlen(str));
384  return ptr == nullptr ? -1l : ptr - vector.data();
385  }

387  protected:
388  Vector<uint8_t> vector{0};
389  size_t available_byte_count = 0;
