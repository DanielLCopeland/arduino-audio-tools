419 template<typename T>
420 size_t readSamples(Stream* p_stream, T* data, int samples){
421  uint8_t *p_result = (uint8_t*) data;
422  int open = samples*sizeof(T);
423  int total = 0;
424  int count0 = 0;
425  // copy missing data - we abort after 5 x read returned 0 bytes
426  while (open > 0){
427  int read = p_stream->readBytes(p_result+total, open);
428  open -= read;
429  total += read;
430  if (read == 0){
431  count0++;
432  delay(1);
433  } else {
434  count0 = 0;
435  }
436  // abort loop
437  if (count0 > 5)
438  break;
439  }
440  // convert bytes to samples
441  return total / sizeof(T);
442 }
