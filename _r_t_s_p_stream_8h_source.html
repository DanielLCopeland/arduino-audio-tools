282  "t=0 0\r\n" // start / stop - 0 -> unbounded and permanent session
283  "m=audio 0 RTP/AVP 98\r\n" // UDP sessions with format 98=aptx

321 class RTSPFormatPCM : public RTSPFormatAudioTools {
322  public:
324  const char* format(char* buffer, int len) override {
325  TRACEI();
326  snprintf(buffer, len,
327  "s=Microphone\r\n" // Stream Name
328  "c=IN IP4 0.0.0.0\r\n" // Connection Information
329  "t=0 0\r\n" // start / stop - 0 -> unbounded and permanent session
330  "m=audio 0 RTP/AVP %d\r\n" // UDP sessions with format 10 or 11
331  "a=rtpmap:%s\r\n"
332  "a=rate:%i\r\n", // provide sample rate
333  format(cfg.channels), payloadFormat(cfg.sample_rate, cfg.channels),
334  cfg.sample_rate);
335  return (const char*)buffer;
336  }

342  const int format(int channels) {
343  int result = 0;
344  switch (channels) {
345  case 1:
346  result = 11;

348  case 2:

350  break;

353  break;
354  }

365  sampleRate, channels);
