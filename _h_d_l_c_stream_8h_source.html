6 /* HDLC Asynchronous framing */
7 /* The frame boundary octet is 01111110, (7E in hexadecimal notation) */
8 #define FRAME_BOUNDARY_OCTET 0x7E
9 
10 /* A "control escape octet", has the bit sequence '01111101', (7D hexadecimal)
11  */
12 #define CONTROL_ESCAPE_OCTET 0x7D
13 
14 /* If either of these two octets appears in the transmitted data, an escape
15  * octet is sent, */
16 /* followed by the original data octet with bit 5 inverted */
17 #define INVERT_OCTET 0x20

19 /* The frame check sequence (FCS) is a 16-bit CRC-CCITT */
20 /* AVR Libc CRC function is _crc_ccitt_update() */
21 /* Corresponding CRC function in Qt (www.qt.io) is qChecksum() */
22 #define CRC16_CCITT_INIT_VAL 0xFFFF

132  int charReceiver(uint8_t data) {
133  /* FRAME FLAG */
134  bool result = 0;
135  if (data == FRAME_BOUNDARY_OCTET) {
136  if (this->has_escape_character == true) {
137  this->has_escape_character = false;
138  }
139  /* If a valid frame is detected */
140  else if ((this->frame_position >= 2) &&
141  (this->frame_checksum ==
142  ((this->frame_buffer.data()[this->frame_position - 1] << 8) |
143  (this->frame_buffer.data()[this->frame_position - 2] &
144  0xff)))) // (msb << 8 ) | (lsb & 0xff)
145  {
146  /* Call the user defined function and pass frame to it */
147  result = this->frame_position - 2;
148  }
149  this->frame_position = 0;
150  this->frame_checksum = CRC16_CCITT_INIT_VAL;
151  return result;
152  }
153 
154  if (this->has_escape_character) {
155  this->has_escape_character = false;
156  data ^= INVERT_OCTET;
157  } else if (data == CONTROL_ESCAPE_OCTET) {
158  this->has_escape_character = true;
159  return 0;
160  }
161 
162  frame_buffer.data()[this->frame_position] = data;
163 
164  if (this->frame_position - 2 >= 0) {
165  this->frame_checksum = _crc_ccitt_update(
166  this->frame_checksum, frame_buffer.data()[this->frame_position - 2]);
167  }
168 
169  this->frame_position++;
170 
171  if (this->frame_position == this->max_frame_length) {
172  this->frame_position = 0;
173  this->frame_checksum = CRC16_CCITT_INIT_VAL;
174  }
175  return 0;

179  void sendFrame(const uint8_t *framebuffer, uint8_t frame_length) {
180  uint8_t data;
181  uint16_t fcs = CRC16_CCITT_INIT_VAL;
182 
183  p_out->write((uint8_t)FRAME_BOUNDARY_OCTET);
184 
185  while (frame_length) {
186  data = *framebuffer++;
187  fcs = HDLCStream::_crc_ccitt_update(fcs, data);
188  if ((data == CONTROL_ESCAPE_OCTET) || (data == FRAME_BOUNDARY_OCTET)) {
189  p_out->write((uint8_t)CONTROL_ESCAPE_OCTET);
190  data ^= INVERT_OCTET;
191  }
192  p_out->write((uint8_t)data);
193  frame_length--;
194  }
195  data = low(fcs);
196  if ((data == CONTROL_ESCAPE_OCTET) || (data == FRAME_BOUNDARY_OCTET)) {
197  p_out->write((uint8_t)CONTROL_ESCAPE_OCTET);
198  data ^= (uint8_t)INVERT_OCTET;
199  }
200  p_out->write((uint8_t)data);
201  data = high(fcs);
202  if ((data == CONTROL_ESCAPE_OCTET) || (data == FRAME_BOUNDARY_OCTET)) {
203  p_out->write(CONTROL_ESCAPE_OCTET);
204  data ^= INVERT_OCTET;
205  }
206  p_out->write(data);
207  p_out->write(FRAME_BOUNDARY_OCTET);
208  }
