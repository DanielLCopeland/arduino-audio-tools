168  size_t writeExpandChannel(const void *src, size_t size_bytes) {
169  size_t writtenBytes = 0;
170  switch(cfg.bits_per_sample){
171  case 8:
172  for(int i = 0; i<size_bytes - 1; i += 2){//2 Samples written at a time to write 4 bytes
173  int8_t frame[4];
174  int8_t *data = (int8_t*) src;
175  frame[0] = data[i];
176  frame[1] = data[i];
177  frame[2] = data[i+1];//i<size_bytes-1 in for() because of this +1
178  frame[3] = data[i+1];
179  bool justWritten = i2s.write(*(uint32_t*)frame, true);//blocking
180  if(justWritten){
181  writtenBytes += 2;
182  } else return writtenBytes;
183  }
184  break;
185  case 16:
186  for(int i = 0; i<size_bytes/sizeof(int16_t); i++){//1 sample written at a time to write 4 bytes
187  int16_t frame[2];
188  int16_t *data = (int16_t *) src;
189  frame[0] = data[i];
190  frame[1] = data[i];
191  bool justWritten = i2s.write(*(uint32_t*)frame, true);//blocking
192  if(justWritten){
193  writtenBytes += sizeof(int16_t);
194  } else return writtenBytes;
195  }
196  break;
197  case 24://24bps are already stored as left-aligned int32_t => handle just like 32bps
198  case 32:
199  for(int i = 0; i<size_bytes/sizeof(int32_t); i++){//1 sample written at a time to write 2*4 bytes
200  int32_t frame[2];
201  int32_t *data = (int32_t *) src;
202  frame[0] = data[i];
203  frame[1] = data[i];
204  bool justWritten = i2s.write(frame[0], true) && i2s.write(frame[1], true);
205  if(justWritten){
206  writtenBytes += sizeof(int32_t);
207  } else return writtenBytes;
208  }
209  }
210  return writtenBytes;
211  }
